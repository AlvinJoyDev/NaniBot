const Discord = require("discord.js");
const { DiscordTogether } = require('discord-together');
client.discordTogether = new DiscordTogether(client);
exports.execute = async (client, message, args) => {
if(message.member.voice.channel) {
            client.discordTogether.createTogetherCode(message.member.voice.channel.id, 'youtube').then(async invite => {
		const suembed = new Discord.MessageEmbed()
		.setTitle(`Successfully started Youtube Together`)
		.setURL(invite.code)
		.setColor('RANDOM')
	return message.inlineReply(suembed);
    });
} else return message.inlineReply('<a:error:858895117916962846> You must be in a voice channel to use that command!')
}
exports.help = {
  name: "youtube",
  aliases: ['yt','youtubetogether'],
  category: 'Fun',
  usage: `youtube`
};
